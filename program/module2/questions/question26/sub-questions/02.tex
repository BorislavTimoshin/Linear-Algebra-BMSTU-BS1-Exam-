\subsection{
    След линейного оператора.
}

\begin{definition}
    \textbf{\textit{Следом линейного оператора $\mathscr{A}$ (матрицы $A$)}} называется сумма диагональных элементов матрицы $A$ линейного оператора $\mathscr{A}$.
\end{definition}

\begin{designation}
    $\tr \mathscr{A}$ или $\Sp \mathscr{A}$.
\end{designation}
